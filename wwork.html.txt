    <script>
        const baseDate = new Date(2026, 3, 1, 12, 0, 0); // 2026/4/1 正午
        
        const shiftCycle = [
            { name: "日勤", class: "text-blue-600 bg-blue-50/50 border-blue-100", label: "日勤", canWork: true, defaultShift: "Bのみ" },
            { name: "夜勤", class: "text-purple-600 bg-purple-50/50 border-purple-100", label: "夜勤", canWork: false, defaultShift: null },
            { name: "明け", class: "text-emerald-600 bg-emerald-50/50 border-emerald-100", label: "明け", canWork: true, defaultShift: "BorC" },
            { name: "休み", class: "text-slate-500 bg-slate-100 border-slate-200", label: "休み", canWork: false, defaultShift: null }
        ];

        let currentYear = 2026;
        let currentMonth = 3; 
        let isScreenshotMode = false;

        const bodyEl = document.getElementById('bodyEl');
        const mainCard = document.getElementById('mainCard');
        const controlPanel = document.getElementById('controlPanel');
        const yearSelect = document.getElementById('yearSelect');
        const monthSelect = document.getElementById('monthSelect');
        const listTitle = document.getElementById('listTitle');
        const possibleDaysGrid = document.getElementById('possibleDaysGrid');
        const matchCount = document.getElementById('matchCount');
        const copyTextarea = document.getElementById('copyTextarea');
        const copyBtn = document.getElementById('copyBtn');
        const copyAlert = document.getElementById('copyAlert');
        const prevMonthBtn = document.getElementById('prevMonthBtn');
        const nextMonthBtn = document.getElementById('nextMonthBtn');
        const screenshotToggleBtn = document.getElementById('screenshotToggleBtn');
        const toggleText = document.getElementById('toggleText');
        const toggleIcon = document.getElementById('toggleIcon');

        function initSelectors() {
            for (let y = 2026; y <= 2030; y++) {
                const opt = document.createElement('option');
                opt.value = y;
                opt.textContent = `${y}年`;
                yearSelect.appendChild(opt);
            }
            for (let m = 0; m < 12; m++) {
                const opt = document.createElement('option');
                opt.value = m;
                opt.textContent = `${m + 1}月`;
                monthSelect.appendChild(opt);
            }
            yearSelect.value = currentYear;
            monthSelect.value = currentMonth;
        }

        function getShiftForDate(date) {
            const d1 = new Date(baseDate.getFullYear(), baseDate.getMonth(), baseDate.getDate(), 12, 0, 0);
            const d2 = new Date(date.getFullYear(), date.getMonth(), date.getDate(), 12, 0, 0);
            const diffTime = d2.getTime() - d1.getTime();
            const diffDays = Math.round(diffTime / (1000 * 60 * 60 * 24));
            const cycleIndex = ((diffDays % 4) + 4) % 4;
            return { ...shiftCycle[cycleIndex], index: cycleIndex };
        }

        function getWWorkAvailability(date, shift) {
            const dayOfWeek = date.getDay();
            if (dayOfWeek === 5 || dayOfWeek === 6) {
                return { available: false, shiftType: null };
            }
            if (shift.index === 0) {
                return { available: true, shiftType: "Bのみ" };
            }
            if (shift.index === 2) {
                return { available: true, shiftType: "BorC" };
            }
            return { available: false, shiftType: null };
        }

        function render() {
            possibleDaysGrid.innerHTML = '';
            listTitle.textContent = `${currentYear}年 ${currentMonth + 1}月`;
            const lastDayOfMonth = new Date(currentYear, currentMonth + 1, 0, 12, 0, 0);
            const totalDays = lastDayOfMonth.getDate();
            const possibleDays = [];

            for (let day = 1; day <= totalDays; day++) {
                const currentDate = new Date(currentYear, currentMonth, day, 12, 0, 0);
                const shift = getShiftForDate(currentDate);
                const avail = getWWorkAvailability(currentDate, shift);
                if (avail.available) {
                    possibleDays.push({ date: currentDate, shiftType: avail.shiftType, mainShift: shift.name });
                }
            }
            updateGrid(possibleDays);
        }

        function updateGrid(days) {
            matchCount.textContent = `${days.length}日間`;
            if (days.length === 0) {
                possibleDaysGrid.className = "block text-center py-8 text-slate-400";
                possibleDaysGrid.innerHTML = `<p class="text-xs font-medium">可能な日がありません。</p>`;
                copyTextarea.value = "勤務可能な日はありません。";
                return;
            }
            possibleDaysGrid.className = "grid grid-cols-2 gap-1.5 text-xs";
            let copyText = `■ ${currentYear}年${currentMonth + 1}月 ダブルワーク可能日\n`;

            days.forEach(item => {
                const dayOfWeekStr = ["日", "月", "火", "水", "木", "金", "土"][item.date.getDay()];
                const isBOnly = item.shiftType === "Bのみ";
                const badgeClass = isBOnly ? "bg-blue-600 text-white" : "bg-emerald-600 text-white";
                const badgeLabel = isBOnly ? "Bのみ" : "B / C可";

                const cell = document.createElement('div');
                cell.className = "flex items-center justify-between p-2 rounded-lg border border-slate-100 bg-slate-50/80 shadow-sm gap-1";
                cell.innerHTML = `
                    <div class="flex items-center gap-1.5 min-w-0">
                        <div class="font-bold text-slate-700 shrink-0">
                            ${item.date.getDate()}<span class="text-[9px] font-normal text-slate-400">(${dayOfWeekStr})</span>
                        </div>
                        <div class="text-[9px] px-1 py-0.5 rounded bg-white border border-slate-200 text-slate-500 shrink-0">
                            ${item.mainShift[0]}
                        </div>
                    </div>
                    <span class="text-[9px] font-black px-1.5 py-0.5 rounded-md ${badgeClass} shrink-0">
                        ${badgeLabel}
                    </span>
                `;
                possibleDaysGrid.appendChild(cell);
                copyText += `${currentMonth + 1}/${item.date.getDate()}(${dayOfWeekStr}): ${item.shiftType === "Bのみ" ? "Bのみ" : "B/C可"}\n`;
            });
            copyTextarea.value = copyText;
        }

        screenshotToggleBtn.addEventListener('click', () => {
            isScreenshotMode = !isScreenshotMode;
            if (isScreenshotMode) {
                bodyEl.classList.add('screenshot-mode');
                bodyEl.classList.remove('bg-slate-50');
                bodyEl.classList.add('bg-white');
                mainCard.classList.remove('shadow-sm', 'border-slate-200');
                mainCard.classList.add('border-transparent');
                controlPanel.classList.add('hidden');
                toggleText.textContent = "戻す";
                toggleIcon.textContent = "🔄";
                screenshotToggleBtn.className = "text-[11px] bg-indigo-600 text-white font-bold py-1 px-2.5 rounded-lg border border-indigo-600 transition-colors flex items-center gap-1";
            } else {
                bodyEl.classList.remove('screenshot-mode');
                bodyEl.classList.add('bg-slate-50');
                bodyEl.classList.remove('bg-white');
                mainCard.classList.add('shadow-sm', 'border-slate-200');
                mainCard.classList.remove('border-transparent');
                controlPanel.classList.remove('hidden');
                toggleText.textContent = "スクショ用";
                toggleIcon.textContent = "📸";
                screenshotToggleBtn.className = "text-[11px] bg-slate-100 hover:bg-indigo-50 text-slate-600 hover:text-indigo-600 font-bold py-1 px-2.5 rounded-lg transition-colors flex items-center gap-1 border border-slate-200";
            }
        });

        yearSelect.addEventListener('change', (e) => { currentYear = parseInt(e.target.value); render(); });
        monthSelect.addEventListener('change', (e) => { currentMonth = parseInt(e.target.value); render(); });
        prevMonthBtn.addEventListener('click', () => {
            currentMonth--;
            if (currentMonth < 0) { currentMonth = 11; currentYear--; if (currentYear < 2026) { currentYear = 2026; currentMonth = 0; } }
            yearSelect.value = currentYear; monthSelect.value = currentMonth; render();
        });
        nextMonthBtn.addEventListener('click', () => {
            currentMonth++;
            if (currentMonth > 11) { currentMonth = 0; currentYear++; if (currentYear > 2030) { currentYear = 2030; currentMonth = 11; } }
            yearSelect.value = currentYear; monthSelect.value = currentMonth; render();
        });
        copyBtn.addEventListener('click', () => {
            copyTextarea.select();
            document.execCommand('copy');
            copyAlert.classList.remove('hidden');
            setTimeout(() => { copyAlert.classList.add('hidden'); }, 2000);
        });

        window.onload = function() { initSelectors(); render(); };
    </script>
